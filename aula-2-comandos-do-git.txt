COMANDOS BÁSICOS DO GIT:

1- cd e o nome da pasta. 'entrar em uma pasta'
2- cd .. 'sair de uma pasta'
3- clear 'limpar a tela'
4- git init 'cria um repositório na pasta atual'
5- git status 'mostra o status atual do repositório'
6- git add e o nome do arquivo 'ele adiciona o arquivo para o seu controle de versão'
7- git add *.txt 'adiciona todos os arquivos txt'
8- git add *.php 'adiciona todos os arquivos php'
9- git add *.css 'adiciona todos os arquivos css'
10- git add . 'adiciona todos arquivos'
11- git commit 'salva as mudanças no projeto'
12- git diff 'mostra o que foi alterado no arquivo'
13- git diff --staged 'mostra o que está na staging area que vai ser enviado para o proximo commitj'
14- git log 'mostra o log de todos commit que você fez no projeto'
15- gitk 'abre o visualizador de alterações do git'
16- git log -p 'Coloca os commit em ordem decrecente e mostra o que foi alterado em cada arquivo'
17- git log --pretty=oneline 'Serve para mostra o codigo de cada commit e a mensagem desse commit'
18- git commit --amend -m 'Faz alteração na mensagem do commit sem precisar criar um novo'
19- git checkout -- 'nome do arquivo', 'Descarta mudanças no arquivo e ele volta ao status original'
20- git rm 'nome do arquivo', 'Remove o arquivo'
21- git tag 'Mostra as tags que você criou'
22- git tag -a 'nome da tag' -m 'Nome da tag elaborado'
23- git show 'e o nome da tag', 'Mostra quem fez a tag a data e hora e outras informações'
24- git tag -d 'nome da tag', 'Deleta a tag'
